1. Create Database
CREATE DATABASE mylibrary;

2. Create Table
- genres
CREATE table genres( id integer PRIMARY KEY AUTO_INCREMENT, name varchar(255) NOT null );

- movies
CREATE TABLE movies ( id integer PRIMARY KEY AUTO_INCREMENT, title varchar(255) NOT null, summary text NOT null, stock integer not null, genre_id integer NOT null, FOREIGN KEY(genre_id) REFERENCES genres(id) );
